# 23. Merge k Sorted Lists

Merge *k* sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

**Example:**

```
Input:
[
  1->4->5,
  1->3->4,
  2->6
]
Output: 1->1->2->3->4->4->5->6
```

```python
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def mergeKLists(self, lists):
        """
        :type lists: List[ListNode]
        :rtype: ListNode
        """
        if len(lists) == 0:
            return None
        
        res = []
        for node in lists:
            while node:
                heapq.heappush(res, node.val)
                node = node.next
            
        dummy = cur = ListNode(-1)
        while res:
            cur.next = ListNode(heapq.heappop(res))
            cur = cur.next
        return dummy.next
      
class Solution2(object):
    def mergeKLists(self, lists):
        """
        :type lists: List[ListNode]
        :rtype: ListNode
        """
        if len(lists) == 0:
            return None
        return self.mergeSort(lists, 0, len(lists)-1)
        
    def mergeSort(self, lists, start, end):
        if start < end:
            mid = start + (end - start) // 2
            l1 = self.mergeSort(lists, start, mid)
            l2 = self.mergeSort(lists, mid + 1, end)
            return self.merge2Lists(l1, l2)
        else:
            return lists[start]
            
    def merge2Lists(self, l1, l2):
        if not l1 or not l2:
            return l1 or l2
        
        dummy = cur = ListNode(-1)
        while l1 and l2:
            if l1.val < l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        
        if l1 or l2:
            cur.next = l1 or l2
        
        return dummy.next
```
